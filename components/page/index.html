Usage

jsx
import { StarOutlined } from '@ant-design/icons';
import { Button, Input, Page, Space } from '@yuntijs/ui';
import { Link } from 'react-router';

const { Breadcrumb, Header, Content } = Page;

const IndexPage = () => {
  return (
    <Page Link={Link}>
      <Breadcrumb items={[{ title: '', path: '/plugins' }, { title: '' }] />
      <Header
        descriptions={[
          {
            icon: {
              content: <UserOutlined />,
              tooltip: '',
            },
            text: '',
          },
        ]}
        extraContent={{
          items: [
            {
              key: 'delete',
              label: '',
              danger: true,
            },
          ],
          onClick: (key, e) => console.log('key', key, e),
        }}
        icon={'https://zos.alipayobjects.com/rmsportal/jkjgkEfvpUPVyRjUImniVslZfWPnJuuZ.png'}
        status={{
          status: 'processing',
          text: '',
        }}
        subTitle=""
        title=""
      />
      <Content>
        <Space size={12}>
          <Button icon={<PlusOutlined />} type="primary">
            
          </Button>
          <Button icon={<ReloadOutlined />}></Button>
          <Input.Search placeholder="" />
        </Space>
        {/* ... */}
      </Content>
    </Page>
  );
};
export default IndexPage;
